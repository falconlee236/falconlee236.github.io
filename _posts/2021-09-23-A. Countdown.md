```cpp
#include <iostream>
#include <string>
using namespace std;

int main(){
    int t; cin >> t;
    while(t--){
        int n; cin >> n;
        char arr[100];
        int ans = 0;
        for(int i = 0; i < n; i++) cin >> arr[i];
        for(int i = n - 1; i >= 0; i--){
            ans += (arr[i] - '0');
            if(i != n - 1 && arr[i] != '0') ans++;
        }
        cout << ans << "\n";
    }
}
```
